# Goto
  ctrl-]        - Goto definition
  ctrl-minus    - return
  cmd-shift-O   - (cmd-r in st) Goto method
  cmd-shift-.   - breadcrumbs (similar to cmd-r)
  ctrl-l         - `zz` scroll cursor location to viewport mid

# Multi-selection
  cmd-D         - next (cmd-u undo last)
  cmd-shift-L   - ALL  (cmd-shift-G)
  opt-shift-I   - split into lines (cmd-shift-L in st)

# Line operations
  shift-opt-up/down  - duplicate
  opt-up/down        - move line

# Focus
  cmd-1         - Editor
  cmd-B         - Sidebar # hide/show
  cmd-shift-E   - File Explore (toggle to Editor)

  cmd-J         - Panel # hide/show
  cmd-shift-Y   - Debug console
  cmd-shift-U   - Output
  cmd-shift-M   - Problems
    terminal ctrl-` (shift for new) cmd-[ cmd-]

  ctrl-cmd-left / ctrl-cmd-right split Editor
  cmd-p ctrl-enter to remove preview on file

# Trim whitespace
  cmd-K cmd-X
